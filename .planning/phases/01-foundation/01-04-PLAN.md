```csharp
    wakeVM.LoadScheme(GetCurrentSchemeId(FeatureType.ScreenWake));
    ConfigurationContent = wakeVM;
    break;
```

**Add Helper Method:**
```csharp
private string GetCurrentSchemeId(FeatureType featureType)
{
    var activeScheme = SchemesByFeature[featureType].FirstOrDefault(s => s.IsActive);
    return activeScheme?.Id ?? string.Empty;
}
```
