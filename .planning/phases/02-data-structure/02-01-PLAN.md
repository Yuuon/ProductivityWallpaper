```csharp
public class ThemeService
{
    private readonly ILogger<ThemeService>? _logger;
    private readonly JsonSerializerOptions _jsonOptions;
    
    public ThemeService(ILogger<ThemeService>? logger = null)
    {
        _logger = logger;
        _jsonOptions = new JsonSerializerOptions
        {
            WriteIndented = true,
            PropertyNamingPolicy = JsonNamingPolicy.CamelCase,
            DefaultIgnoreCondition = JsonIgnoreCondition.WhenWritingNull
        };
    }
    
    // Load theme from folder
    public async Task<ThemeManifest?> LoadAsync(string themeFolderPath, bool validateResources = true)
    
    // Save theme to folder
    public async Task SaveAsync(ThemeManifest manifest, string themeFolderPath)
    
    // Create new empty theme
    public ThemeManifest CreateNew(string name, string author)
    
    // Validation
    public bool Validate(ThemeManifest manifest, string themeFolderPath, out List<string> errors)
    
    // Resource management helpers
    public Task<string> AddResourceAsync(ThemeManifest manifest, string sourceFilePath, string themeFolderPath)
    public Task RemoveResourceAsync(ThemeManifest manifest, string resourceId, string themeFolderPath)
}
```

Implementation notes:
- Load: Read theme.json, deserialize to ThemeManifest, optionally validate all resource files exist
- Save: Ensure folders exist (/images, /videos, /audio, /thumbnails), write theme.json
- AddResource: Copy file to appropriate subfolder, generate thumbnail (placeholder for now), add ResourceEntry
- RemoveResource: Delete files, remove from library, remove references from schemes (cascade)
