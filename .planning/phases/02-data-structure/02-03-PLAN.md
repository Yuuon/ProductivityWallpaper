```csharp
/// <summary>
/// Represents a user-defined anniversary or countdown event.
/// Stored in UserWidgetSettings, not in ThemeManifest.
/// </summary>
public partial class AnniversaryEventModel : ObservableObject
{
    /// <summary>
    /// Unique identifier for this event.
    /// </summary>
    [ObservableProperty]
    private string _id = Guid.NewGuid().ToString();
    
    /// <summary>
    /// Display name of the event.
    /// </summary>
    [ObservableProperty]
    private string _name = string.Empty;
    
    /// <summary>
    /// Target date for the event.
    /// </summary>
    [ObservableProperty]
    private DateTime _targetDate;
    
    /// <summary>
    /// Repeat mode for recurring events.
    /// </summary>
    [ObservableProperty]
    private RepeatMode _repeatMode = RepeatMode.NoRepeat;
    
    /// <summary>
    /// Optional: Time of day for the event (if applicable).
    /// </summary>
    [ObservableProperty]
    private TimeSpan? _targetTime;
    
    /// <summary>
    /// Optional: Description or notes.
    /// </summary>
    [ObservableProperty]
    private string _description = string.Empty;
    
    /// <summary>
    /// Whether this event is enabled.
    /// </summary>
    [ObservableProperty]
    private bool _isEnabled = true;
    
    /// <summary>
    /// Calculates the days remaining until the event.
    /// </summary>
    public int GetDaysRemaining()
    {
        var today = DateTime.Today;
        var target = GetNextOccurrence();
        return (target - today).Days;
    }
    
    /// <summary>
    /// Gets the next occurrence date based on repeat mode.
    /// </summary>
    public DateTime GetNextOccurrence()
    {
        var today = DateTime.Today;
        var target = TargetDate.Date;
        
        if (RepeatMode == RepeatMode.NoRepeat)
            return target;
        
        // If target is in the future, return it
        if (target > today)
            return target;
        
        // Calculate next occurrence based on repeat mode
        return RepeatMode switch
        {
            RepeatMode.Yearly => GetNextYearlyOccurrence(target, today),
            RepeatMode.Monthly => GetNextMonthlyOccurrence(target, today),
            RepeatMode.Weekly => GetNextWeeklyOccurrence(target, today),
            _ => target
        };
    }
    
    private DateTime GetNextYearlyOccurrence(DateTime target, DateTime today)
    {
        var next = new DateTime(today.Year, target.Month, target.Day);
        if (next < today)
            next = next.AddYears(1);
        return next;
    }
    
    private DateTime GetNextMonthlyOccurrence(DateTime target, DateTime today)
    {
        var next = new DateTime(today.Year, today.Month, target.Day);
        if (next < today)
            next = next.AddMonths(1);
        return next;
    }
    
    private DateTime GetNextWeeklyOccurrence(DateTime target, DateTime today)
    {
        var daysUntilTarget = ((int)target.DayOfWeek - (int)today.DayOfWeek + 7) % 7;
        return today.AddDays(daysUntilTarget == 0 ? 7 : daysUntilTarget);
    }
}

/// <summary>
/// Repeat mode for anniversary events.
/// </summary>
public enum RepeatMode
{
    NoRepeat,
    Yearly,
    Monthly,
    Weekly
}
```

```csharp
/// <summary>
/// Service for loading and saving user-specific widget settings.
/// Settings are stored in user_config.json alongside AppConfig.
/// </summary>
public class UserSettingsService
{
    private const string SettingsFileName = "user_config.json";
    private readonly string _settingsPath;
    private readonly JsonSerializerOptions _jsonOptions;
    private UserWidgetSettings? _cachedSettings;
    
    public UserSettingsService()
    {
        _settingsPath = Path.Combine(
            Environment.GetFolderPath(Environment.SpecialFolder.ApplicationData),
            "ProductivityWallpaper",
            SettingsFileName);
        
        _jsonOptions = new JsonSerializerOptions
        {
            WriteIndented = true,
            PropertyNamingPolicy = JsonNamingPolicy.CamelCase,
            DefaultIgnoreCondition = JsonIgnoreCondition.WhenWritingNull
        };
    }
    
    /// <summary>
    /// Loads user widget settings. Creates defaults if file doesn't exist.
    /// </summary>
    public async Task<UserWidgetSettings> LoadAsync()
    {
        if (_cachedSettings != null)
            return _cachedSettings;
        
        if (File.Exists(_settingsPath))
        {
            try
            {
                var json = await File.ReadAllTextAsync(_settingsPath);
                var config = JsonSerializer.Deserialize<AppConfigWithWidgets>(json, _jsonOptions);
                if (config?.WidgetSettings != null)
                {
                    _cachedSettings = config.WidgetSettings;
                    return _cachedSettings;
                }
            }
            catch (Exception ex)
            {
                // Log error, fall through to create defaults
                Debug.WriteLine($"Error loading user settings: {ex.Message}");
            }
        }
        
        // Create defaults
        _cachedSettings = CreateDefaultSettings();
        await SaveAsync(_cachedSettings);
        return _cachedSettings;
    }
    
    /// <summary>
    /// Saves user widget settings.
    /// </summary>
    public async Task SaveAsync(UserWidgetSettings settings)
    {
        _cachedSettings = settings;
        
        // Load existing config to preserve other settings
        AppConfigWithWidgets config;
        if (File.Exists(_settingsPath))
        {
            var json = await File.ReadAllTextAsync(_settingsPath);
            config = JsonSerializer.Deserialize<AppConfigWithWidgets>(json, _jsonOptions) 
                ?? new AppConfigWithWidgets();
        }
        else
        {
            config = new AppConfigWithWidgets();
        }
        
        config.WidgetSettings = settings;
        
        // Ensure directory exists
        Directory.CreateDirectory(Path.GetDirectoryName(_settingsPath)!);
        
        var newJson = JsonSerializer.Serialize(config, _jsonOptions);
        await File.WriteAllTextAsync(_settingsPath, newJson);
    }
    
    /// <summary>
    /// Gets the cached settings without reloading.
    /// </summary>
    public UserWidgetSettings? GetCachedSettings() => _cachedSettings;
    
    /// <summary>
    /// Invalidates the cache, forcing a reload on next access.
    /// </summary>
    public void InvalidateCache() => _cachedSettings = null;
    
    private UserWidgetSettings CreateDefaultSettings()
    {
        return new UserWidgetSettings
        {
            ClockSettings = new ClockWidgetSettings(),
            PomodoroSettings = new PomodoroWidgetSettings(),
            AnniversarySettings = new AnniversaryWidgetSettings(),
            GlobalPomodoroSettings = new PomodoroGlobalSettings(),
            UsePerThemePomodoroSettings = false
        };
    }
    
    /// <summary>
    /// Internal class to wrap widget settings alongside existing AppConfig.
    /// </summary>
    private class AppConfigWithWidgets
    {
        public string? LastLibraryPath { get; set; }
        public string? MediaLibraryPath { get; set; }
        public UserWidgetSettings? WidgetSettings { get; set; }
    }
}
```

Key design:
1. Stores settings in user_config.json alongside existing AppConfig
2. Caches settings in memory to avoid repeated file I/O
3. Creates sensible defaults if settings don't exist
4. Preserves existing config when saving
